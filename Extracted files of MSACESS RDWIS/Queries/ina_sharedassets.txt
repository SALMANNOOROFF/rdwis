QUERY: ina_sharedassets

SQL:
SELECT ina_invats.ias_desc, ina_invats.ias_details, ina_invatcomps.iac_qty, ina_invatcomps.iac_qtyunit, ina_invats.ias_unt_id, ina_invatcomps.iac_location
FROM ina_invats INNER JOIN ina_invatcomps ON ina_invats.ias_id = ina_invatcomps.iac_ias_id
WHERE (((ina_invatcomps.iac_shared)="1"))
ORDER BY ina_invats.ias_price DESC;
