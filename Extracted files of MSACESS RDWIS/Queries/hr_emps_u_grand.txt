QUERY: hr_emps_u_grand

SQL:
SELECT hr_emps_u.*, hr_empsexta.*, hr_empsextb.*, hr_empsextc.*, units.*, hr_contracts_last.*, hr_bnkaccountsforpay.*, hr_vehiclesconcat.*, hr_qualifs_max.qlf_name, hr_qualifs_max.qlf_grade, grades.grade_id, Concatrelated("job_company","hr_jobs","job_emp_id='" & [emp_id] & "'") AS experience, grades.grade AS grade_full, IntervalInMonths([ctr_startdt],[ctr_enddt]) AS contract_period, namecomplete([emp_name],[emp_title],[emp_rank]) AS name_full, IIf([ctr_type]=1,"Full Time","Part Time") AS contract_type
FROM ((((((((hr_emps_u LEFT JOIN hr_contracts_last ON hr_emps_u.emp_id = hr_contracts_last.ctr_num) LEFT JOIN units ON hr_emps_u.emp_unt_id = units.unt_id) LEFT JOIN hr_empsexta ON hr_emps_u.emp_id = hr_empsexta.empexta_emp_id) LEFT JOIN hr_empsextb ON hr_emps_u.emp_id = hr_empsextb.empextb_emp_id) LEFT JOIN hr_empsextc ON hr_emps_u.emp_id = hr_empsextc.empextc_emp_id) LEFT JOIN hr_bnkaccountsforpay ON hr_emps_u.emp_id = hr_bnkaccountsforpay.bac_emp_id) LEFT JOIN hr_vehiclesconcat ON hr_emps_u.emp_id = hr_vehiclesconcat.vcl_emp_id) LEFT JOIN hr_qualifs_max ON hr_emps_u.emp_id = hr_qualifs_max.qlf_emp_id) LEFT JOIN grades ON hr_contracts_last.ctr_grade = grades.grade_short;
