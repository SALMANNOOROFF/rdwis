QUERY: hr_emps_qualifwise_doc

SQL:
TRANSFORM Count(hr_empsexta.empexta_emp_id) AS CountOfempexta_emp_id
SELECT qualifications.qualification_name, qualifications.qualification_id, Count(hr_empsexta.empexta_emp_id) AS Total
FROM ((hr_emps_u_active INNER JOIN hr_empsexta ON hr_emps_u_active.emp_id = hr_empsexta.empexta_emp_id) INNER JOIN qualifications ON hr_empsexta.emp_qualif = qualifications.qualification_id) INNER JOIN units ON hr_emps_u_active.emp_unt_id = units.unt_id
WHERE (((qualifications.qualification_name)<>"Course"))
GROUP BY qualifications.qualification_name, qualifications.qualification_id
ORDER BY qualifications.qualification_id DESC 
PIVOT units.unt_namesh In ("Comm","Enab","NWS","Sensors","SoSE","Sys","Fin","HR","Admin","IT","IS");
