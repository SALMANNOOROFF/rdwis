SQL:
SELECT First(fin_commitments.cmt_docid) AS cmt_docid, First(fin_commitments.cmt_type) AS cmt_type, fin_docs_shd.subhead, fin_commitments.cmt_id, First(Round([cmt_amount]*[ratio],2)) AS cmt_amountz, Round(Sum(IIf([trn_transtype]=1,Nz([trn_amount1],0),Nz([trn_amount2],0)*[ratio])),2) AS amount, First(fin_commitments.cmt_effhed_id) AS cmt_effhed_id, First(fin_commitments.cmt_hed_id) AS cmt_hed_id, First(fin_transactions.trn_id) AS trn_id
FROM (fin_commitments LEFT JOIN fin_transactions ON fin_commitments.cmt_id = fin_transactions.trn_cmt_id) LEFT JOIN fin_docs_shd ON (fin_commitments.cmt_docid = fin_docs_shd.doc_id) AND (fin_commitments.cmt_type = fin_docs_shd.doc_type)
WHERE (((fin_commitments.cmt_hed_id)=Forms!vars!Parameter1) And ((fin_commitments.cmt_type) In ("Ps","Pt","Rb","Sa")) And ((fin_commitments.cmt_status)="Awaited"))
GROUP BY fin_docs_shd.subhead, fin_commitments.cmt_id;
