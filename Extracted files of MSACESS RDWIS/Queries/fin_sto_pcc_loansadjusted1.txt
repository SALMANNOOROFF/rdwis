QUERY: fin_sto_pcc_loansadjusted1

SQL:
SELECT fin_loanadjustments.lad_from, fin_loanadjustments.lad_to, IIf([lad_from]=[HeadId],1,-1)*[lad_amount] AS amount, [HeadId] AS hed_id
FROM fin_loanadjustments
WHERE (((fin_loanadjustments.lad_from)=[HeadId])) OR (((fin_loanadjustments.lad_to)=[HeadId]));
