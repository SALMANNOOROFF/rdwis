Public Function NextStop(strDoc As String) As Integer
Dim intCurrentLevel As Integer
Dim rstNext As Recordset
Dim strNext As String
Dim str As String
Dim intloc1 As Integer
Dim intloc2 As Integer

intCurrentLevel = getVar("varRoleLevel")
Set rstNext = CurrentDb.OpenRecordset("Select gvar_value From cen_globalvars Where gvar_name = '" & strDoc & "'", dbOpenForwardOnly)
strNext = rstNext!gvar_value
intloc1 = InStr(1, strNext, intCurrentLevel)
intloc1 = InStr(intloc1 + 1, strNext, "-")
intloc2 = InStr(intloc1 + 1, strNext, "-")
str = Mid(strNext, intloc1 + 1, intloc2 - intloc1 - 1)
If str Like "(*)" Then str = Mid(str, 2, Len(str) - 2)
NextStop = CInt(str)
End Function

Public Function PrevStop(strDoc As String) As Integer
Dim intCurrentLevel As Integer
Dim rstPrev As Recordset
Dim strPrev As String
Dim str As String
Dim intloc1 As Integer
Dim intloc2 As Integer

intCurrentLevel = getVar("varRoleLevel")
Set rstPrev = CurrentDb.OpenRecordset("Select gvar_value From cen_globalvars Where gvar_name = '" & strDoc & "'", dbOpenForwardOnly)
strPrev = rstPrev!gvar_value
intloc1 = InStr(1, strPrev, intCurrentLevel)
intloc1 = InStrRev(strPrev, "-", intloc1)
intloc2 = InStrRev(strPrev, "-", intloc1 - 1)
str = Mid(strPrev, intloc2 + 1, intloc1 - intloc2 - 1)
If str Like "(*)" Then str = Mid(str, 2, Len(str) - 2)
PrevStop = CInt(str)
End Function


Public Function FirstStop(strDoc As String) As Integer
Dim rstFirst As Recordset
Dim strFirst As String
Dim str As String
Dim intloc As Integer

Set rstFirst = CurrentDb.OpenRecordset("Select gvar_value From cen_globalvars Where gvar_name = '" & strDoc & "'", dbOpenForwardOnly)
strFirst = rstFirst!gvar_value
intloc = InStr(2, strFirst, "-")
str = Mid(strFirst, 2, intloc - 2)
If str Like "(*)" Then str = Mid(str, 2, Len(str) - 2)
FirstStop = CInt(str)
End Function

Public Function LastStop(strDoc As String) As Integer
Dim rstLast As Recordset
Dim strLast As String
Dim str As String
Dim intloc As Integer

Set rstLast = CurrentDb.OpenRecordset("Select gvar_value From cen_globalvars Where gvar_name = '" & strDoc & "'", dbOpenForwardOnly)
strLast = rstLast!gvar_value
intloc = InStrRev(strLast, "-", Len(strLast) - 1)
str = Mid(strLast, intloc + 1, Len(strLast) - intloc - 1)

Public Function JunctionStop(strDoc As String) As Integer
Dim rstJunction As Recordset
Dim strJunction As String
Dim intloc1 As Integer
Dim intloc2 As Integer

Set rstJunction = CurrentDb.OpenRecordset("Select gvar_value From cen_globalvars Where gvar_name = '" & strDoc & "'", dbOpenForwardOnly)
strJunction = rstJunction!gvar_value
intloc1 = InStrRev(strJunction, "(")
intloc2 = InStrRev(strJunction, ")")
If intloc1 = 0 Or intloc2 = 0 Then
    JunctionStop = 0
    Else
    JunctionStop = Mid(strJunction, intloc1 + 1, intloc2 - intloc1 - 1)
    End If

End Function

Public Function IsStop(strDoc As String) As Boolean
Dim rstIsStop As Recordset
Dim strIsStop As String

IsStop = False
Set rstIsStop = CurrentDb.OpenRecordset("Select gvar_value From cen_globalvars Where gvar_name = '" & strDoc & "'", dbOpenForwardOnly)
strIsStop = rstIsStop!gvar_value
If InStr(strIsStop, "-" & getVar("varRoleLevel") & "-") Then IsStop = True
If InStr(strIsStop, "-(" & getVar("varRoleLevel") & ")-") Then IsStop = True
End Function

Public Function ReceivedFrom(strRoute As String) As Integer
Dim intloc1 As Integer
Dim intloc2 As Integer
Dim str As String

intloc1 = InStrRev(strRoute, "-", Len(strRoute) - 1)
intloc2 = InStrRev(strRoute, "-", intloc1 - 1)
str = Mid(strRoute, intloc2 + 1, intloc1 - intloc2 - 1)
If str Like "(*)" Then str = Mid(str, 2, Len(str) - 2)
ReceivedFrom = CInt(str)
End Function

Public Function DesigInStop(Level As Integer) As String
Dim dbsIn As Database
Dim rstIn As Recordset

Set dbsIn = CurrentDb()
Set rstIn = dbsIn.OpenRecordset("Select rol_desigshort from cen_roles Where rol_level = " & Level & " And rol_unt_id = 180000", dbOpenSnapshot)
If rstIn.EOF Then
    DesigInStop = ""
    Else
    DesigInStop = rstIn!rol_desigshort
    End If
End Function

Public Function MprWith(Level As Integer) As String
Select Case Level
    Case FirstStop("mpr_route"): MprWith = "Initiator"
    Case 25: MprWith = " SO R&D"
    Case Else: MprWith = DesigInStop(Level)
    End Select
End Function

Public Function SemiStatus(ProgLevel As Variant, Progstatus As Variant) As String
Dim intThisLevel As Integer

ProgLevel = Nz(ProgLevel, FirstStop("mpr_route"))
Progstatus = Nz(Progstatus, "NG")
intThisLevel = getVar("varRoleLevel")


If Progstatus = "Finalized" Then
    SemiStatus = "Finalized"
    Exit Function
    End If
    
Select Case ProgLevel
    Case Is > intThisLevel
        SemiStatus = "Forwarded"
    Case Is = intThisLevel
        SemiStatus = "Action Awaited"
    Case Is < intThisLevel
        SemiStatus = "Receipt Awaited"
    End Select

End Function
