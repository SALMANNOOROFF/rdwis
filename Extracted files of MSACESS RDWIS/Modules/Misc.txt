Public Sub AllowEditsAdvanced(FormName As String, Optional AllowAddInSubform, Optional AllowDelInSubform)
Dim ctl As Control

Forms(FormName).AllowEdits = True
For Each ctl In Forms(FormName).Controls
    Select Case ctl.ControlType
        Case acTextBox, acComboBox, acListBox
            If ctl.Tag Like "*l*" Then
                ctl.Locked = True
                Else
                ctl.BorderStyle = 1
                If ctl.ControlType = acComboBox Then Forms(FormName)("box_" & ctl.name).Visible = False
                End If
        Case acSubform
            If Not ctl.Tag Like "*l*" Then
                If Not IsMissing(AllowAddInSubform) Then
                    If CBool(AllowAddInSubform) = True Then ctl.Form.AllowAdditions = True
                    Else
                    If ctl.Tag Like "*a*" Then ctl.Form.AllowAdditions = True
                    End If
                
                If Not IsMissing(AllowDelInSubform) Then
                    If CBool(AllowDelInSubform) = True Then
                        ctl.Form.AllowDeletions = True
                        ctl.Form.RecordSelectors = True
                        End If
                    Else
                    If ctl.Tag Like "*d*" Then
                        ctl.Form.AllowDeletions = True
                        ctl.Form.RecordSelectors = True
                        End If
                    End If
                AllowEditsAdvancedsub FormName, ctl.name
                End If
        End Select
        
    Next ctl

End Sub

Public Sub AllowEditsAdvancedsub(FormName2 As String, SubFormName As String)
Dim ctl2 As Control
On Error GoTo Error_Control
Forms(FormName2)(SubFormName).Form.AllowEdits = True
For Each ctl2 In Forms(FormName2)(SubFormName).Form.Controls
    Select Case ctl2.ControlType
        Case acTextBox, acComboBox
            If ctl2.Tag Like "*l*" Then
                ctl2.Locked = True
                Else
                ctl2.BorderStyle = 1
                If ctl2.ControlType = acComboBox Then Forms(FormName2)(SubFormName).Form("box_" & ctl2.name).Visible = False
                End If
        End Select
    Next ctl2
Exit Sub

Error_Control:
If Err.Number = 2455 Then
    Exit Sub
    Else
    MsgBox Err.Number & " - " & Err.Description
    End If
End Sub

Public Sub OpenNotepad()
On Error GoTo Error_Handler

Dim strFormName As String
Dim strSubContainer As String
Dim strControlName As String
Dim booAllowEdits As Boolean

strFormName = Screen.ActiveForm.name
strSubContainer = Screen.ActiveForm.ActiveControl.name
strControlName = Screen.ActiveControl.name
booAllowEdits = Screen.ActiveForm.AllowEdits And Not Screen.ActiveControl.Locked And Screen.ActiveControl.Enabled

DoCmd.OpenForm "notepad"
If strSubContainer = strControlName Then
    Forms!notepad.txtNote = Forms(strFormName).Controls(strControlName).Text
    Else
    Forms!notepad.txtNote = Forms(strFormName).Controls(strSubContainer).Form.Controls(strControlName).Text
    End If
If booAllowEdits = True Then Forms!notepad.AllowEdits = True
Forms!notepad.txtNote.SelLength = 0
Forms!notepad.txtFormName = strFormName
Forms!notepad.txtSubContainer = strSubContainer
Forms!notepad.txtControlName = strControlName
Forms!notepad.AllowEdits = booAllowEdits

Exit Sub
Error_Handler:
If Err.Number = 2452 Then
    Resume Next
    Else
    MsgBox Error$
    End If
End Sub

Public Sub CreateMprGroupIfRequired()
Dim dbsMpr As Database
Dim rstMpr As Recordset
Dim strLook As String

strLook = Format(LastDatePrevMonth(Date), "mmm yy")
Set dbsMpr = CurrentDb()
Set rstMpr = dbsMpr.OpenRecordset("Select * from prj_mprgroup Where mgp_name = '" & strLook & "'", dbOpenSnapshot)
If Not rstMpr.EOF Then Exit Sub

Set rstMpr = dbsMpr.OpenRecordset("Select * from prj_mprgroup Where mgp_status = 'Current'")
rstMpr.Edit
rstMpr!mgp_status = "Old"
rstMpr.Update
rstMpr.AddNew
rstMpr!mgp_name = strLook
rstMpr!mgp_dtg = GetNow()
rstMpr!mgp_status = "Current"
rstMpr.Update
    
End Sub

Public Function NewIdForProjectOrAccount(UnitId As Long) As Long
Dim dbsId As Database
Dim rstId As Recordset
Dim lngLowerLimit As Long
Dim lngUpperLimit As Long
Dim strSql As String

Set dbsId = CurrentDb()
'Get limits
strSql = "SELECT unt_lowers, unt_uppers FROM cen_units WHERE unt_id = " & UnitId
Set rstId = dbsId.OpenRecordset(strSql, dbOpenSnapshot)
lngLowerLimit = rstId!unt_lowers
lngUpperLimit = rstId!unt_uppers
rstId.Close

'Calculate Number
strSql = "SELECT prj_id As num FROM prj_projects UNION " & _
         "SELECT hed_id As num FROM cen_heads WHERE hed_type = 'Project'"
strSql = "SELECT num FROM (" & strSql & ") WHERE num > " & lngLowerLimit & _
         " AND num <= " & lngUpperLimit
strSql = "SELECT Max(num) AS max_num FROM (" & strSql & ")"
Set rstId = dbsId.OpenRecordset(strSql, dbOpenSnapshot)
NewIdForProjectOrAccount = Nz(rstId!max_num, lngLowerLimit) + 1
End Function

Function MakeRef(ControlName As String) As String
Dim F As Form
Dim R As Report
Dim C1 As Control, c2 As Control
On Error Resume Next

Set F = Screen.ActiveForm
Set R = Screen.ActiveReport
If F Is Nothing Then
    Select Case R.name
        Case "hr_ctrcasehgminutes": Set F = Forms("hr_ctrcaseminutes")
        End Select
    End If

Set C1 = F.Controls(ControlName & "_encl")
Set c2 = F.Controls(ControlName & "_flag")
MakeRef = "Encl " & C1.Value & " (F/" & c2.Value & ")"
End Function

Public Sub StartWait()
Dim n As Integer
DoCmd.Hourglass True
DoCmd.OpenForm "wait"
For n = 1 To 50
    DoEvents
    Next
End Sub

Public Sub EndWait()
DoCmd.Close acForm, "wait"
DoCmd.Hourglass False
End Sub
