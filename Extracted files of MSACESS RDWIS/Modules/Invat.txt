Function ChargeOffItem(CompId As Long, HeldQty As Double, ChargeOffQty As Double, DispDate As Date, CompStatus As String, CompRemarks As String) As String
Dim wrkCoff As Workspace
Dim dbsCoff As Database
Dim rstCoff As Recordset
Dim strCommit As String
Dim strCoff As String
On Error GoTo Error_Handler

Set wrkCoff = DBEngine.Workspaces(0)
Set dbsCoff = CurrentDb()

'Charge-off qty = held qty
If ChargeOffQty = HeldQty Then
    wrkCoff.BeginTrans
    strCommit = "Begun"
    
    Set rstCoff = dbsCoff.OpenRecordset("Select * From ina_invatcomps where iac_id = " & CompId & " Or iac_parent_id = " & CompId, dbOpenDynaset)
    Do While Not rstCoff.EOF
        rstCoff.Edit
        rstCoff!iac_status = CompStatus
        rstCoff.Update
        strCoff = strCoff & rstCoff!iac_ias_id & " (" & rstCoff!iac_qty & " " & rstCoff!iac_qtyunit & ")" & vbCrLf
        rstCoff.MoveNext
        Loop
    
    wrkCoff.CommitTrans
    strCommit = "Committed"
    ChargeOffItem = "Item charged off" & vbCrLf & vbCrLf & strCoff
    Exit Function
    End If

'Charge-off qty < held qty
CompId = SplitItem(CompId, ChargeOffQty)
Set rstCoff = dbsCoff.OpenRecordset("Select * From ina_invatcomps where iac_id = " & CompId, dbOpenDynaset)
rstCoff.Edit
rstCoff!iac_qty = ChargeOffQty
rstCoff!iac_dtg = GetNow()
rstCoff!iac_dispdate = DispDate
rstCoff!iac_dispdtg = GetNow()
rstCoff!iac_status = CompStatus
rstCoff!iac_remarks = CompRemarks
rstCoff.Update
ChargeOffItem = "Item charged off" & vbCrLf & vbCrLf & rstCoff!iac_ias_id & " (" & rstCoff!iac_qty & " " & rstCoff!iac_qty & ")"

Exit Function
Error_Handler:
If strCommit = "Begun" Then wrkCoff.Rollback
MsgBox Err.Number & " - " & Err.Description
End Function

Function AttachItem(ChildInvatId As Long, ChildId As Long, ChildHeld As Double, ChildQty As Double, ParentId As Long, ParentHeld As Long) As Long
Dim dbsAttach As Database
Dim rstAttach As Recordset
Dim lngSimChild As Long
On Error GoTo Error_Handler

Set dbsAttach = CurrentDb()

'Parent
If ParentHeld > 1 Then ParentId = SplitItem(ParentId, 1)
Set rstAttach = dbsAttach.OpenRecordset("Select * From ina_invatcomps where iac_id = " & ParentId)
If rstAttach!iac_isparent = "0" Then
    rstAttach.Edit
    rstAttach!iac_isparent = "1"
    rstAttach.Update
    End If
rstAttach.Close

'Get Similar child id
Set rstAttach = dbsAttach.OpenRecordset("Select iac_id From ina_invatitems where iac_status = 'Held' And " & _
                                        "iac_parent_id = " & ParentId & " And ias_id = " & ChildInvatId, dbOpenSnapshot)
If Not rstAttach.EOF Then lngSimChild = rstAttach!iac_id
rstAttach.Close

'Child
Select Case True
    Case ChildQty = ChildHeld And lngSimChild = 0
        Set rstAttach = dbsAttach.OpenRecordset("Select * From ina_invatcomps where iac_id = " & ChildId)
        rstAttach.Edit
        rstAttach!iac_parent_id = ParentId
        rstAttach.Update
        rstAttach.Close

    Case ChildQty < ChildHeld And lngSimChild = 0
        ChildId = SplitItem(ChildId, ChildQty)
        Set rstAttach = dbsAttach.OpenRecordset("Select * From ina_invatcomps where iac_id = " & ChildId)
        rstAttach.Edit
        rstAttach!iac_parent_id = ParentId
        rstAttach.Update
        rstAttach.Close
        
    Case ChildHeld = ChildQty And lngSimChild <> 0
        CombineItems lngSimChild, ChildId, ChildQty

    Case ChildHeld > ChildQty And lngSimChild <> 0
        Set rstAttach = dbsAttach.OpenRecordset("Select * From ina_invatcomps where iac_id = " & ChildId)
        rstAttach.Edit
        rstAttach!iac_qty = rstAttach!iac_qty - ChildQty
        rstAttach.Update
        rstAttach.Close
        Set rstAttach = dbsAttach.OpenRecordset("Select iac_qty From ina_invatcomps where iac_id = " & lngSimChild)
        rstAttach.Edit
        rstAttach!iac_qty = rstAttach!iac_qty + ChildQty
        rstAttach.Update
        rstAttach.Close

    End Select
AttachItem = ParentId
    
Exit Function
Error_Handler:
MsgBox Err.Number & " - " & Err.Description
End Function

Function DetachItem(ChildInvatId As Long, ChildId As Long, ChildHeld As Double, ChildQty As Double, ParentId As Long, ParentInvatId As Long) As Long
Dim dbsDetach As Database
Dim rstDetach As Recordset
Dim lngSimSolo As Long
On Error GoTo Error_Handler

Set dbsDetach = CurrentDb()

'Get Similar solo id for child
Set rstDetach = dbsDetach.OpenRecordset("Select iac_id From ina_invatitems where iac_status = 'Held' And ias_id = " & ChildInvatId & " And " & _
                                        "Nz([iac_parent_id],'')='' And iac_isparent = '0'", dbOpenSnapshot)
If Not rstDetach.EOF Then lngSimSolo = rstDetach!iac_id
rstDetach.Close

'Child
Select Case True
    Case ChildQty = ChildHeld And lngSimSolo = 0
        MsgBox "equal and no other solo"
        Set rstDetach = dbsDetach.OpenRecordset("Select * From ina_invatcomps where iac_id = " & ChildId)
        rstDetach.Edit
        rstDetach!iac_parent_id = Null
        rstDetach.Update
        rstDetach.Close

    Case ChildQty < ChildHeld And lngSimSolo = 0
        MsgBox "less and no other solo"
        ChildId = SplitItem(ChildId, ChildQty)
        Set rstDetach = dbsDetach.OpenRecordset("Select * From ina_invatcomps Where iac_id = " & ChildId)
        rstDetach.Edit
        rstDetach!iac_parent_id = Null
        rstDetach.Update
        rstDetach.Close

    Case ChildHeld = ChildQty And lngSimSolo <> 0
        MsgBox "equal with other solo"
        CombineItems lngSimSolo, ChildId, ChildQty

    Case ChildQty < ChildHeld And lngSimSolo <> 0
        MsgBox "less with other solo"
        Set rstDetach = dbsDetach.OpenRecordset("Select * From ina_invatcomps where iac_id = " & ChildId)
        rstDetach.Edit
        rstDetach!iac_qty = rstDetach!iac_qty - ChildQty
        rstDetach.Update
        rstDetach.Close
        Set rstDetach = dbsDetach.OpenRecordset("Select iac_qty From ina_invatcomps where iac_id = " & lngSimSolo)
        rstDetach.Edit
        rstDetach!iac_qty = rstDetach!iac_qty + ChildQty
        rstDetach.Update
        rstDetach.Close

    End Select
lngSimSolo = 0

'See if parent is no more a parent
Set rstDetach = dbsDetach.OpenRecordset("Select iac_id From ina_invatitems where iac_status = 'Held' And iac_parent_id = " & ParentId, dbOpenSnapshot)
If rstDetach.EOF Then
    'Get similar solo id for parent
    Set rstDetach = dbsDetach.OpenRecordset("Select iac_id From ina_invatitems where iac_status = 'Held' And ias_id = " & ParentInvatId & " And " & _
                                        "Nz([iac_parent_id],'')='' And iac_isparent = '0'", dbOpenSnapshot)
    If Not rstDetach.EOF Then lngSimSolo = rstDetach!iac_id
    Select Case lngSimSolo
        Case 0
            Set rstDetach = dbsDetach.OpenRecordset("Select * From ina_invatcomps where iac_id = " & ParentId)
            rstDetach.Edit
            rstDetach!iac_isparent = "0"
            rstDetach.Update
            DetachItem = ParentId
        Case Else
            CombineItems lngSimSolo, ParentId, 1
            DetachItem = lngSimSolo
        End Select
    rstDetach.Close
    End If

Exit Function
Error_Handler:
MsgBox Err.Number & " - " & Err.Description
End Function


Function SplitItem(CmpId As Long, SplitQty As Double) As Long
Dim wrkSplt As Workspace
Dim dbsSplt As Database
Dim rstSplt As Recordset
Dim strCmt As String
Dim arrData() As Variant
Dim intFields As Integer
Dim n As Integer
On Error GoTo Error_Handler

Set wrkSplt = DBEngine.Workspaces(0)
Set dbsSplt = CurrentDb()
Set rstSplt = dbsSplt.OpenRecordset("Select * From ina_invatcomps where iac_id = " & CmpId, dbOpenDynaset, dbSeeChanges)

wrkSplt.BeginTrans
strCmt = "Begun"

'Reduce qty in current record
rstSplt.Edit
rstSplt!iac_qty = rstSplt!iac_qty - SplitQty
rstSplt.Update

'Get field values in an array
intFields = rstSplt.Fields.count - 1
ReDim arrData(1 To intFields)
For n = 1 To intFields
    arrData(n) = rstSplt.Fields(n)
    Next n

'Add New record and populate
rstSplt.AddNew
For n = 1 To intFields
    If Nz(arrData(n), "") <> "" Then rstSplt.Fields(n) = arrData(n)
    Next n
rstSplt!iac_qty = SplitQty
rstSplt.Update
rstSplt.Bookmark = rstSplt.LastModified

wrkSplt.CommitTrans
strCmt = "Committed"
SplitItem = rstSplt!iac_id

Exit Function
Error_Handler:
If strCmt = "Begun" Then wrkSplt.Rollback
MsgBox Err.Number & " - " & Err.Description
End Function

Sub CombineItems(Cmp1Id As Long, Cmp2Id As Long, CombineQty As Double)
Dim wrkCmb As Workspace
Dim dbsCmb As Database
Dim rstCmb As Recordset
Dim strCmt As String
On Error GoTo Error_Handler

Set dbsCmb = CurrentDb
Set wrkCmb = DBEngine.Workspaces(0)
wrkCmb.BeginTrans
strCmt = "Begun"

'Add quantity in first record
Set rstCmb = dbsCmb.OpenRecordset("Select iac_qty From ina_invatcomps where iac_id = " & Cmp1Id)
rstCmb.Edit
rstCmb!iac_qty = rstCmb!iac_qty + CombineQty
rstCmb.Update
rstCmb.Close

'Delete second record
Set rstCmb = dbsCmb.OpenRecordset("Select iac_id FROM ina_invatcomps WHERE iac_id = " & Cmp2Id)
rstCmb.Edit
rstCmb.Delete
rstCmb.Close

wrkCmb.CommitTrans
strCmt = "Committed"

Exit Sub
Error_Handler:
If strCmt = "Begun" Then
    wrkCmb.Rollback
    MsgBox "Combine operation rolled back." & Err.Number & " - " & Err.Description
    Else
    MsgBox Err.Number & " - " & Err.Description
    End If
End Sub

Sub CreateAssembly(ItemDesc As String, ItemDetails As String, ItemType As Integer, ProjectID As Long, UnitId As Long)
On Error GoTo Error_Handler
Dim wrkAssm As Workspace
Dim dbsAssm As Database
Dim rstAssm As Recordset
Dim strCmt As String
Dim dtgDtg As Date
Dim lngIasId As Long

Set wrkAssm = DBEngine.Workspaces(0)
Set dbsAssm = CurrentDb()
wrkAssm.BeginTrans
strCmt = "Begun"


Set rstAssm = dbsAssm.OpenRecordset("ina_invats", dbOpenDynaset, dbSeeChanges)
dtgDtg = GetNow()
rstAssm.AddNew
rstAssm!ias_desc = ItemDesc
'rstAssm!ias_details = ItemDetails
rstAssm!ias_type = 7
rstAssm!ias_type2 = ItemType
rstAssm!ias_qty = 1
rstAssm!ias_qtyunit = "num"
rstAssm!ias_prj_id = ProjectID
rstAssm!ias_unt_id = UnitId
rstAssm!ias_dtg = dtgDtg
rstAssm!ias_chargedate = dtgDtg
rstAssm.Update
rstAssm.Bookmark = rstAssm.LastModified
lngIasId = rstAssm!ias_id
rstAssm.Close

Set rstAssm = dbsAssm.OpenRecordset("ina_invatcomps")
rstAssm.AddNew
rstAssm!iac_ias_id = lngIasId
rstAssm!iac_details = ItemDetails
rstAssm!iac_qty = 1
rstAssm!iac_qtyunit = "num"
rstAssm!iac_isassembly = "1"
rstAssm!iac_status = "Held"
rstAssm!iac_dtg = dtgDtg
rstAssm.Update
rstAssm.Close

wrkAssm.CommitTrans
strCmt = "Committed"

Exit Sub
Error_Handler:
If strCmt = "Begun" Then
    wrkAssm.Rollback
    MsgBox "Assembly creation rolled back." & Err.Number & " - " & Err.Description
    Else
    MsgBox Err.Number & " - " & Err.Description
    End If

End Sub
